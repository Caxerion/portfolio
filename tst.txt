        // Check authentication status
        (function() {

            const authTime = sessionStorage.getItem('authTime');
            const currentTime = Date.now();
            
            // If not authenticated or session expired (1 hour), redirect to auth
            if (!isAuth || !authTime || (currentTime - parseInt(authTime)) > 3600000) {

                sessionStorage.removeItem('authTime');

                return;
            }
            
            // Update auth time for session extension

        })();
